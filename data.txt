     function send(){
         // $('#signupForm').submit();
         //1201(进口水果)
         if($("#prodCatid").val()==="1201"){
             if($("#pubDateStartTime").val()!="") {
                 if($("#pubDateStartTime").val()<"2023/01/01") {
                     $("#pubDateStartTime").val("2023/01/01")
                     $('#demo19').val("2023/01/01");
                 }
             }
         }
         if($('#demo19').val()===""){
             $("#pubDateStartTime").val("2023/01/01")
             $('#demo19').val("2023/01/01");
         }

         var param = {
               "limit" : $("#limit").val(),
               "current" : $("#current").val(),
               "pubDateStartTime" : $("#pubDateStartTime").val(),
               "pubDateEndTime" : $("#pubDateEndTime").val(),
               "prodPcatid" : $("#prodPcatid").val(),
               "prodCatid" : $("#prodCatid").val(),
               "prodName" : $("#prodName").val()
         }

         $.post("/getPriceData.html", param, function (res) {
             $("#count").val(res.count);
             $("#limit").val(res.limit);
             $("#current").val(res.current);

             $("#tableBody").find("tr").remove();

             for (var i = 0; i < res.list.length; i++) {
                 var data = res.list[i];
                 $("#tableBody").append("<tr>" +
                     "<td>" + (data.prodCat == null ? "" : data.prodCat) + "</td>" +
                     "<td>" + (data.prodPcat == null ? "" : data.prodPcat) + "</td>" +
                     "<td>" + (data.prodName == null ? "" : data.prodName) + "</td>" +
                     "<td>" + data.lowPrice + "</td>" +
                     "<td>" + data.avgPrice + "</td>" +
                     "<td>" + data.highPrice + "</td>" +
                     "<td>" + data.specInfo + "</td>" +
                     "<td>" + (data.place == null ? "" : data.place) + "</td>" +
                     "<td>" + data.unitInfo + "</td>" +
                     "<td>" + dateFormat(data.pubDate) + "</td>" +
                     "</tr>");
             }

             //完整功能
             laypage.render({
                 elem: 'page'
                 ,count: $("#count").val()
                 ,limit: $("#limit").val()
                 ,curr: $("#current").val()
                 ,limits: [10, 20]   //每页条数的选择项
                 ,theme: '#1A8C44'
                 ,layout: ['count', 'prev', 'page', 'next', 'limit', 'refresh', 'skip']
                 ,jump: function(obj){
                     if (Number($('#current').val()) != Number(obj.curr) || Number($('#limit').val()) != Number(obj.limit)) {
                         $('#current').val(obj.curr);
                         $('#limit').val(obj.limit);
                         send();
                     }
                 }
             });
         });
     }

     function dateFormat(date) {
         if (date == null)
             return "";

         if (date.length > 10) {
             return date.substring(0, 10);
         } else {
             return date;
         }
     }
